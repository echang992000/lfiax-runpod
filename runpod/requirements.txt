# Pin the JAX/lfiax stack used by sir.py / cli-anything-lfiax on a RunPod
# CUDA pod. The local boed_agent_3 venv was on jax 0.7.1, but jax's
# cu12 plugin pinning on 0.7.x is fragile on fresh pip installs (the
# `[cuda12]` extra sometimes can't find a matching `jax-cuda12-plugin`
# wheel). Pin to 0.4.34 instead: same overall behavior on this stack,
# clean cu12 plugin chain, no resolver headaches.

# JAX with CUDA 12 GPU plugin. Pinned to 0.4.27 — the last 0.4.x
# release whose plugin expects cuDNN 8.9 at runtime (matching what
# torch 2.1.2 ships). 0.4.28+ bumped to cuDNN 9.x, which collides with
# torch's pin and triggers ``CUDNN_STATUS_INTERNAL_ERROR`` on the
# first JAX call. The [with-cuda] extra is intentionally omitted so
# the plugin uses the system / torch-installed CUDA libs instead of
# pulling its own pip-managed copies.
jax==0.4.27
jaxlib==0.4.27
jax-cuda12-plugin==0.4.27
jax-cuda12-pjrt==0.4.27
ml_dtypes>=0.4.0,<0.5

# Sampling / flow stack used by sir.py — versions that import cleanly
# under jax 0.4.34.
blackjax==1.2.1
distrax==0.1.5
dm-haiku==0.0.13
optax==0.2.4
chex==0.1.90
tensorflow_probability
tensorflow
tensorflow_datasets

# torch must match a CUDA build available on RunPod images; the +cu121
# wheel is what the standard PyTorch CUDA 12.1 RunPod template ships
# with. Override at install time if your pod uses a different CUDA.
torch==2.1.2
torchsde==0.2.6

# Hydra + wandb + simulation-based inference tools that sir.py imports.
hydra-core==1.3.2
hydra-submitit-launcher==1.2.0
omegaconf==2.3.0
wandb
sbi==0.26.1

# General scientific stack pinned to versions our local venv used.
numpy<2
matplotlib
tqdm
